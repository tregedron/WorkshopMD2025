#-----------------------------------------------------------------------------
#----------------------------- General setting -------------------------------
#-----------------------------------------------------------------------------
clear
units metal
dimension 3
boundary p p p
atom_style atomic

#-----------------------------------------------------------------------------
#---------------------------- Importing atoms --------------------------------
#-----------------------------------------------------------------------------
read_data cubic56.data
# replicate 2 2 2 

#-----------------------------------------------------------------------------
#-------------------- Grouping for MSD calculations---------------------------
#-----------------------------------------------------------------------------
group F type 1
group Li type 2
group Na type 3
group K type 4 

#-----------------------------------------------------------------------------
#------------------------- Interatomic Potential -----------------------------
#-----------------------------------------------------------------------------
pair_style mlip mlip.ini
pair_coeff * *

#-----------------------------------------------------------------------------
#----------------------------- Other Settings --------------------------------
#-----------------------------------------------------------------------------
variable T equal 1000
variable P equal 0
timestep 0.001
thermo 100
neighbor 2.0 bin 
neigh_modify delay 1 check yes 

#-----------------------------------------------------------------------------
#----------------------------- Output Settings -------------------------------
#-----------------------------------------------------------------------------  
thermo_style custom step etotal pe ke temp enthalpy vol press density

dump mydump all custom 100 lammps.dump id type x y z vx vy vz
dump_modify mydump sort id

#-----------------------------------------------------------------------------
#--------------------------- Equilibration run -------------------------------
#-----------------------------------------------------------------------------  
velocity all create 1400 123456 dist gaussian
fix equil all npt temp 1400 $T 0.1 iso $P $P 1
run 5000
unfix equil

#-----------------------------------------------------------------------------
#---------------------------- Output Settings --------------------------------
#-----------------------------------------------------------------------------
compute msdLi Li msd com yes
compute msdK K msd com yes
compute msdF F msd com yes
compute msdNa Na msd com yes

thermo_style custom step etotal pe ke temp enthalpy vol press density c_msdLi[4] c_msdK[4] c_msdF[4] c_msdNa[4]

#-----------------------------------------------------------------------------
#----------------------------- Production run --------------------------------
#-----------------------------------------------------------------------------
fix production all npt temp $T $T 0.1 iso $P $P 1
run 10000
unfix production

print "All done!"
